// ecommerce example

record LocalizedString { // scalar
    string locale;
    string text;
}
record Color {
    string reference;
    LocalizedString displayName;
}
record ColorRef {
    string reference;
}
record Shape {
    string reference;
    string displayOrder;
    LocalizedString displayName;
}
record ShapeRef {
    string reference;
}
record Brand {
    string reference;
    LocalizedString name;
    map<string,Shape> shapes;
    map<string,Product> products;
    map<string,Product> components;
}
record BrandRef {
    string reference;
}
record Product {
    string reference;
    LocalizedString name;
    LocalizedString description;
    decimal(12,2) price;
    BrandRef brand;
    ColorRef color;
    ShapeRef shape;
    repeatable BillOfMaterialsLine bom;
}
record ProductRef {
    string reference;
}
record Component {
    string reference;
    LocalizedString name;
}
record ComponentRef {
    string reference;
}
record BillOfMaterialsLine {
    ComponentRef component;
    int32 qty;
    string role;
}
record Repository {
    map<string,Color> colors;
    map<string,Brand> brands;
}
